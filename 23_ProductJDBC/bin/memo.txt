drop sequence prdseq ;
create sequence prdseq
 increment by 1
 start with 1
 minvalue 1;
 
 
drop table products;
create table products(
 id number primary key,
 name varchar2(30), 
 stock number,
 price number,
 category varchar2(30),
 inputdate date default sysdate
);
